I = imread("Penguin.jpg");
figure(1);
imshow(I);
% converting RGB image to Gray Scale image
I2 = rgb2gray(I);
figure(2);
imshow(I2);
% converting Gray Scale image to binary using im2bw function
I3 = im2bw(I2,0.6);
figure(3);
imshow(I3);
% converting Gray Scale image to binary using imbinarize function
I4 = imbinarize(I2,"global"); %otsu method
I5 = imbinarize(I2,"adaptive"); % Adaptive thresholoding method
figure(4);
imshow(I4);
figure(5);
imshow(I5);
