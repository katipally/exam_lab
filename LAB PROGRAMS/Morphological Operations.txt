# Image Morphological operations
I = imread("flowers.jpg");    
I = rgb2grey(I);
figure(1);
imshow(I);  
% Dilated Image 
se = strel("line", 7, 7); 
dilate = imdilate(I, se); 
figure(2);
imshow(dilate); 
title("Dilated image"); 
% Eroded image 
erode = imerode(I, se); 
figure(3);  
imshow(erode); 
title("Eroded image");   
% Opened image 
open = imopen(I, se); 
figure(4);  
imshow(open); 
title("Opened image"); 
% Closed image 
close = imclose(I, se); 
figure(5);  
imshow(close); 
title("Closed image");
