i = imread("n1.jpg");
figure(1);
imshow(i);
%Constructing histogram of an image
figure(2);
imhist(i);
%Contrast streching to enhance the quality of input image
i1 = imadjust(i,[0.1,0.8],[0.0,1]);
figure(3);
imshow(i1);
figure(4);
imhist(i1);
%Contrast streching using histogram equalisation technique 
to enhance the quality of input image
i2 = histeq(i);
i3 = histeq(i,8); %number of quanatas/colors is 8
figure(5);
imshow(i2);
figure(6);
imshow(i3);
%histograms after equalisation
figure(7);
imhist(i2);
figure(8);
imhist(i3);
