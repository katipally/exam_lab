% Image sharpening in spatial domain
clear all;
I = imread("n1.jpg");
I1 = rgb2gray(I);
figure(1);
imshow(I1);
% designing a laplacian mask
m = [0 1 0,1 -4 1,0 1 0];
m1 = [1 1 1,1 -8 1,1 1 1];
m2 = [0 -1 0,-1 4 -1,0 -1 0];
m3 = [-1 -1 -1,-1 8 -1,-1 -1 -1];
% applying filter to extract/highlight edge features
I2 = imfilter(I1,m);
% Adding the filtered resultenent image to original image 
I3 = I1 + I2;
figure(2);
imshow(I3);
