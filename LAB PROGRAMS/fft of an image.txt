clear all;
I = imread("n1.jpg");
I1 = rgb2gray(I);
figure(1);
imshow(I1);
% fft of input image
I2 = fft2(I1);
figure(2);
imshow(mat2gray(log(1+abs(I2))));
impixelinfo % displays the normalised intesity value of pixel coordinates 
% shifted fft of input image 
I3 = fftshift(I2);
figure(3);
imshow(mat2gray(log(1+abs(I3))));
impixelinfo
